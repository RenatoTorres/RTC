16static float getTemp_Datasheet(float targetRatio) {
17 int low = 0;
18 int high = sizeof(TFPT_Vishay_Values) / sizeof(TFPT_Vishay_Values[0]) - 1;
19 int resultIndex = -1;
20
21 while (low <= high) {
22 int mid = low + (high - low) / 2;
23
24 if (TFPT_Vishay_Values[mid].fRatio_R25 == targetRatio) {
25 resultIndex = mid;
26 break;
27 }
28
29 if (TFPT_Vishay_Values[mid].fRatio_R25 < targetRatio) {
30 low = mid + 1;
31 } else {
32 resultIndex = mid;
33 high = mid - 1;
34 }
35 }
36
37 if (resultIndex != -1) {
38 // Found the exact match or the next lower fRatio
39 if (resultIndex > 0) {
40 // Use linearization between 2 datasheet points to optimize accuracy
41 // fTx = fT1 + ((fRx - fR1)/(fR2 - fR1))*(T2 - T1);
42 // (T2 - T1) is always 1 oC
43 float fT1 = TFPT_Vishay_Values[resultIndex - 1].ftemperature;
44 float fT2 = TFPT_Vishay_Values[resultIndex].ftemperature;
45 float fR1 = TFPT_Vishay_Values[resultIndex - 1].fRatio_R25;
46 float fR2 = TFPT_Vishay_Values[resultIndex].fRatio_R25;
47 float fTx = fT1 + ((targetRatio - fR1) / (fR2 - fR1));
48
49 #if( _DEBUG_RTC_API == 1)
50 printf("getTemp_Datasheet(targetRatio=%f, fT1=%f, fT2=%f, fR1=%f, fR2=%f) => fTx = %f\n", targetRatio, fT1, fT2, fR1, fR2, fTx);
51 #endif
52
53 return fTx;
54 }
55 else
56 {
57 #if( _DEBUG_RTC_API == 1)
58 printf("getTemp_Datasheet(targetRatio=%f, fTx=%f\n", targetRatio, TFPT_Vishay_Values[resultIndex].ftemperature);
59 #endif
60
61 return TFPT_Vishay_Values[resultIndex].ftemperature;
62 }
63 } else {
64 // Return 'absolute zero' in case of ratio out of range
65 return -273.0f;
66 }
